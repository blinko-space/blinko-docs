  </Accordion>
  <Accordion  title="Hashtag">
    Add a hashtag to the selected text.
  </Accordion>
  <Accordion title="List Icon">
    For creating unordered (bulleted) or ordered (numbered) lists.
  </Accordion>
  <Accordion title="Checklist Icon">
    Allows the creation of to-do lists or checkboxes for task management.
  </Accordion>
  <Accordion title="Table Icon">
    For inserting tables into the content.
  </Accordion>
  <Accordion title="Image Icon">
    To embed images in the document.
  </Accordion>
  <Accordion title="Code Icon">
    For inserting code snippets into the document.
  </Accordion>
   <Accordion title="Sandpack Icon">
    For embedding sandboxed code editors.
  </Accordion>
  <Accordion title="Link Icon">
    Adds hyperlinks to selected text or inserts a URL.
  </Accordion>
  <Accordion title="Upload Icon">
    Used to upload files or attachments.
  </Accordion>
  <Accordion title="Voice Icon">
    Used to transcribe voice to text.
  </Accordion>
  <Accordion title="Camera Icon">
    Allows capturing images using a connected camera or device.
  </Accordion>
